window.webkit.messageHandlers.WKScriptMessageHandler.postMessage({"funcName":"hehe","body":[]});
body格式:
funName:---->方法名
body:----->参数数组元素必须字符串，没有参数则传[]
{"funcName":"dismiss",
"body":[]
}

window.webkit.messageHandlers.WKScriptMessageHandler.postMessage("");
不符合上述规范

window.webkit.messageHandlers.WKScriptMessageHandler.postMessage();
不符合上述规范
